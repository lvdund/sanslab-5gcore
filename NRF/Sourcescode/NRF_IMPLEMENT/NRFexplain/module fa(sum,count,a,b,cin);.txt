module fa(sum,count,a,b,cin);
input a,b,cin;
output sum,count;
xor x1(w0,a,b);
and a1(w1,a,b);
xor x2(sum,w0,cin);
and a2(w2,w0,cin);
or ol(count,w1,w2);
endmodule
